## The Pattern

Add two nodes to your graph — one to load memory before the agent runs, one to save memory after it responds.

load_memory → agent → save_memory

---

## Full Example

```python
import os
from typing import TypedDict, Annotated
from langgraph.graph import StateGraph, END
from langgraph.graph.message import add_messages
from langchain_openai import ChatOpenAI
from langchain_core.messages import SystemMessage
from remem import AsyncRememClient

memory = AsyncRememClient(api_key=os.getenv("REMEM_API_KEY"))
llm    = ChatOpenAI(model="gpt-4o-mini")
AGENT_ID = "langgraph_agent"


# ── State ─────────────────────────────────────────────────────────

class AgentState(TypedDict):
    messages:       Annotated[list, add_messages]
    user_id:        str
    memory_context: str


# ── Nodes ─────────────────────────────────────────────────────────

async def load_memory_node(state: AgentState) -> AgentState:
    """Load relevant memories before the agent responds."""
    last_message = state["messages"][-1].content

    memories = await memory.recall(
        query=last_message,
        user_id=state["user_id"],
        agent_id=AGENT_ID,
        top_k=5,
        min_score=0.0,
    )

    if memories:
        context = "\n".join(f"- {m.content}" for m in memories)
    else:
        context = "No previous memories for this user."

    state["memory_context"] = context
    return state


async def agent_node(state: AgentState) -> AgentState:
    """Run the LLM with memory context injected."""
    system = SystemMessage(content=f"""You are a helpful assistant.

What you know about this user:
{state['memory_context']}

Use this context in your responses.""")

    response = await llm.ainvoke([system] + state["messages"])
    state["messages"].append(response)
    return state


async def save_memory_node(state: AgentState) -> AgentState:
    """Save the user's message as a memory after responding."""
    last_user_message = [
        m for m in state["messages"]
        if hasattr(m, "type") and m.type == "human"
    ][-1].content

    await memory.remember(
        content=last_user_message,
        user_id=state["user_id"],
        agent_id=AGENT_ID,
        memory_type="episodic",
    )
    return state


# ── Graph ─────────────────────────────────────────────────────────

builder = StateGraph(AgentState)
builder.add_node("load_memory", load_memory_node)
builder.add_node("agent",       agent_node)
builder.add_node("save_memory", save_memory_node)

builder.set_entry_point("load_memory")
builder.add_edge("load_memory", "agent")
builder.add_edge("agent",       "save_memory")
builder.add_edge("save_memory", END)

graph = builder.compile()


# ── Run ───────────────────────────────────────────────────────────

async def chat(user_id: str, message: str) -> str:
    result = await graph.ainvoke({
        "messages":       [{"role": "user", "content": message}],
        "user_id":        user_id,
        "memory_context": "",
    })
    return result["messages"][-1].content


# Test it
import asyncio
asyncio.run(chat("user_123", "My name is Victor and I'm in Lagos"))
asyncio.run(chat("user_123", "What do you know about me?"))
```

## Tips for LangGraph

<AccordionGroup>
  <Accordion title="Use context() instead of recall() at session start">
    `recall()` needs a query. At session start you don't have one yet. Use `context()` to proactively load the most important memories:

```python
    # At session start — no query needed
    ctx = await memory.context(user_id=user_id, agent_id=AGENT_ID, top_k=10)

    # When user sends a message — query-aware retrieval
    mems = await memory.recall(query=user_message, user_id=user_id, agent_id=AGENT_ID)
```
  </Accordion>
  <Accordion title="Don't save every message">
    Filter before saving. Not every message is worth remembering:

```python
    skip_phrases = ["ok", "thanks", "got it", "hi", "hello", "bye"]

    if not any(p in last_message.lower() for p in skip_phrases):
        await memory.remember(last_message, user_id=user_id, agent_id=AGENT_ID)
```
  </Accordion>
</AccordionGroup>
